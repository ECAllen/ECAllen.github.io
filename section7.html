Define the function to interact with Ollama’s local service

def ollama_local_query(prompt: str) -> str: # Initialize Ollama client client = ollama.Client( base_url=“http://localhost:11434” ) # Adjust the base_url to your local server’s URL

# Create a request
request = ollama.GenerateRequest(prompt=prompt)

# Send the request and get the response
response = client.generate(request)

# Extract the text from the response
return response.text
